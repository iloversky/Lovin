/** 
 * 字符串是否以 pattern结尾
 * flag: 是否返回去掉 pattern的字符串
 */
function endWith(source, pattern, flag) {
   if(!flag) {
      if(source.substring(source.length - 1, source.length) == pattern) {
         return true;
      }
      return false;
   } else {
      if(source.substring(source.length - 1, source.length) == pattern) {
         return source.substr(0, source.length - 1);
      }
      return source
   }
}
